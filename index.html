    const spent = data.expenses.filter((e) => e.category === c.name).reduce((s, e) => s + (+e.amount || 0), 0);

    const days = daysInMonth(ym);

    const total = data.habits.length * days;

    if (!total) return 0;

    let done = 0;

    data.habits.forEach((h) => {

      for (let d = 1; d <= days; d++) if (data.habitChecks[`${h.id}-${d}`]) done++;

    });

    return done / total;

  }, [data.habits, data.habitChecks, ym]);


  const goalsDone = data.goals.filter((g) => g.status === 'Готово').length;

  const goalsTotal = data.goals.length;

        <KpiCard label="Привычки" value={fmtPct(habitProgress)} sublabel="среднее" delay={300} />

      </div>

          )}

        </Card>

                    }}

              </div>

            ))}

          </div>

        </Card>

  useEffect(() => {

    if (!scrollerRef.current) return;

    const btn = scrollerRef.current.querySelector(`[data-day="${activeDay}"]`);

    if (btn) btn.scrollIntoView({ behavior: 'smooth', block: 'nearest', inline: 'center' });

  }, [activeDay]);


  const tasks = data.dailyTasks[activeDay] || [];


  const addTask = () => {

    const t = { id: uid(), title: '', status: null, time: '', note: '' };

    update((d) => ({ ...d, dailyTasks: { ...d.dailyTasks, [activeDay]: [...(d.dailyTasks[activeDay] || []), t] } }));

  };

    update((d) => ({ ...d, dailyTasks: { ...d.dailyTasks, [activeDay]: (d.dailyTasks[activeDay] || []).map((t) => t.id === id ? { ...t, ...patch } : t) } }));

    update((d) => ({ ...d, dailyTasks: { ...d.dailyTasks, [activeDay]: (d.dailyTasks[activeDay] || []).filter((t) => t.id !== id) } }));

function Habits({ data, update, ym }) {

  const days = daysInMonth(ym);

  const dayArr = Array.from({ length: days }, (_, i) => i + 1);


  const toggle = (habitId, day) => {

    const key = `${habitId}-${day}`;

    update((d) => ({ ...d, habitChecks: { ...d.habitChecks, [key]: !d.habitChecks[key] } }));

  };

  const addHabit = () => update((d) => ({ ...d, habits: [...d.habits, { id: uid(), name: '' }] }));

  const updateHabit = (id, name) => update((d) => ({ ...d, habits: d.habits.map((h) => h.id === id ? { ...h, name } : h) }));

  const removeHabit = (id) => update((d) => {

    const checks = { ...d.habitChecks };

    Object.keys(checks).forEach((k) => { if (k.startsWith(id + '-')) delete checks[k]; });

    return { ...d, habits: d.habits.filter((h) => h.id !== id), habitChecks: checks };

  });


  const habitStats = (habit) => {

    let done = 0;

    for (let d = 1; d <= days; d++) if (data.habitChecks[`${habit.id}-${d}`]) done++;

    return { done, pct: done / days };

  };


  return (

    <div className="space-y-6">

      <Card>

        <div className="flex items-center justify-between mb-5">

          <SectionTitle>Трекер</SectionTitle>

          <button onClick={addHabit} className="inline-flex items-center gap-1.5 px-3 py-1.5 rounded-lg bg-accent text-white text-xs font-medium hover:opacity-90 transition-opacity">

            <Icon name="plus" className="w-3.5 h-3.5" /> Привычка

          </button>

        </div>


        {data.habits.length === 0 ? (

          <div className="text-center py-12 text-neutral-400 text-sm">Нет привычек</div>

        ) : (

          <div className="overflow-x-auto -mx-6 px-6 hide-scroll">

            <div className="min-w-fit">

              <div className="flex items-center gap-1 mb-2">

                <div className="w-40 flex-shrink-0 text-[10px] font-semibold tracking-[0.14em] text-neutral-500 uppercase pr-3">Привычка</div>

                {dayArr.map((d) => {

                  const dow = dowOfDay(ym, d);

                  return (

                    <div key={d} className={`w-6 h-6 flex flex-col items-center justify-center text-[10px] font-medium ${dow >= 5 ? 'text-amber-600' : 'text-neutral-400'}`}>

                      {d}

                    </div>

                  );

                })}

                <div className="w-12 flex-shrink-0 text-center text-[10px] font-semibold tracking-[0.14em] text-neutral-500 uppercase pl-2">%</div>

              </div>


              {data.habits.map((h, hi) => {

                const stats = habitStats(h);

                return (

function Checklists({ checklists, updateChecklists }) {

  const [checks, setChecks] = useState({});


  const toggleItem = (listId, itemIdx) => {

    const k = `${listId}-${itemIdx}`;

    setChecks((c) => ({ ...c, [k]: !c[k] }));

  };

  const resetList = (listId) => {

    setChecks((c) => {

      const next = { ...c };

      Object.keys(next).forEach((k) => { if (k.startsWith(listId + '-')) delete next[k]; });

      return next;

    });

  };

  const addList = () => updateChecklists((cs) => [...cs, { id: uid(), name: 'Новый список', items: [] }]);

  const renameList = (id, name) => updateChecklists((cs) => cs.map((l) => l.id === id ? { ...l, name } : l));

  const removeList = (id) => updateChecklists((cs) => cs.filter((l) => l.id !== id));

  const addItem = (listId) => updateChecklists((cs) => cs.map((l) => l.id === listId ? { ...l, items: [...l.items, ''] } : l));

  const updateItem = (listId, idx, val) => updateChecklists((cs) => cs.map((l) => {

    if (l.id !== listId) return l;

    const items = [...l.items]; items[idx] = val; return { ...l, items };

  }));

  const removeItem = (listId, idx) => updateChecklists((cs) => cs.map((l) => {

    if (l.id !== listId) return l;

    return { ...l, items: l.items.filter((_, i) => i !== idx) };

  }));


  return (

    <div className="space-y-6">

      <div className="flex justify-end">

        <button onClick={addList} className="inline-flex items-center gap-1.5 px-3 py-2 rounded-lg bg-accent text-white text-sm font-medium hover:opacity-90 transition-opacity">

          <Icon name="plus" className="w-4 h-4" /> Список

        </button>

      </div>


      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">

        {checklists.map((list, li) => {

          const total = list.items.length;

          const done = list.items.reduce((s, _, i) => s + (checks[`${list.id}-${i}`] ? 1 : 0), 0);

          return (

            <Card key={list.id} delay={li * 80}>

              <div className="flex items-start justify-between mb-4">

                <div className="flex-1 min-w-0">

                  <input type="text" value={list.name} onChange={(e) => renameList(list.id, e.target.value)} className="bg-transparent text-lg font-display font-light outline-none w-full" />

                  <div className="text-[10px] tracking-[0.14em] text-neutral-400 uppercase mt-0.5 font-mono">{done} / {total}</div>

                </div>

                <div className="flex">

                  <IconBtn onClick={() => resetList(list.id)} title="Сброс">

                    <Icon name="repeat" className="w-4 h-4" />

                  </IconBtn>

                  <IconBtn onClick={() => removeList(list.id)} title="Удалить" danger>

                    <Icon name="trash" className="w-4 h-4" />

                  </IconBtn>

                </div>

              </div>

              <div className="space-y-1">

                {list.items.map((item, i) => {

                  const k = `${list.id}-${i}`;

                  const on = !!checks[k];

                  return (

                    <div key={i} className="group flex items-center gap-2 py-1">

                      <button onClick={() => toggleItem(list.id, i)} className={`w-5 h-5 rounded-md border flex items-center justify-center flex-shrink-0 transition-all ${on ? 'bg-emerald-600 border-emerald-600' : 'bg-white border-neutral-300 hover:border-neutral-500'}`}>

                        {on && <Icon name="check" className="w-3.5 h-3.5 text-white" strokeWidth={3} />}

                      </button>

                      <input type="text" value={item} onChange={(e) => updateItem(list.id, i, e.target.value)} className={`flex-1 bg-transparent text-sm outline-none transition-all ${on ? 'line-through text-neutral-400' : 'text-neutral-700'}`} />

                      <button onClick={() => removeItem(list.id, i)} className="opacity-0 group-hover:opacity-100 text-neutral-300 hover:text-rose-500 transition-opacity">

                        <Icon name="x" className="w-3.5 h-3.5" />

                      </button>

                    </div>

                  );

                })}

                <button onClick={() => addItem(list.id)} className="text-xs text-neutral-400 hover:text-neutral-700 mt-2 inline-flex items-center gap-1 transition-colors">

      </div>

                for (let i = 0; i < localStorage.length; i++) {

                  const k = localStorage.key(i);

                  if (k.startsWith('planner:')) all[k] = localStorage.getItem(k);

                }

                const blob = new Blob([JSON.stringify(all, null, 2)], { type: 'application/json' });

                const url = URL.createObjectURL(blob);

                const a = document.createElement('a');

                a.href = url;

                a.download = `planner-backup-${new Date().toISOString().slice(0,10)}.json`;

                type="file" accept="application/json" className="hidden"

                onChange={(e) => {

                  const f = e.target.files[0];

                  if (!f) return;

                  const r = new FileReader();

                  r.onload = (ev) => {

                    try {

                      const data = JSON.parse(ev.target.result);

                      Object.entries(data).forEach(([k, v]) => localStorage.setItem(k, v));

                      window.location.reload();

                    } catch (err) { alert('Ошибка импорта: ' + err.message); }

                  };

                  r.readAsText(f);

                }}

              />

            </label>

          </div>

        </div>
